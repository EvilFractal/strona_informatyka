string szyfr(string s,int move, bool szyfruj){
    string wyn="";
    char z;
    move%=26;
    if(!szyfruj) move=26-move;
    for(int i=0;i<s.size();i++){
        z=s[i]+move;
        if(s[i]>='A' and s[i]<='Z'){
            if(z>'Z') z-=26;
        }
        else if(s[i]>='a' and s[i]<='z'){
            if(z>'z') z-=26;
        }
        wyn+=z;
    }
    return wyn;
}
