jednoczesne szukanie wartości maksymalnej i minimalnej na liście

Aby znaleźć wartości minimalną i maksymalną z listy przy tylko jednym przez nią przejściu należy dla każdego jej elementu sprawdzić, czy jest on większy niż dotychczasowo znaleziona wartośc maksymalna lub mniejsza od dotychczas znalezionej wartości minimalnej. 

vector<int> min_and_max(vector<int> v){
    int mini=v[0],maxi=v[0];
    for(int i=1;i<v.size();i++){
        if(v[i]<mini){
            mini=v[i];
        }
        else if(v[i]>maxi){
            maxi=v[i];
        }
    }
    return vector<int> {mini,maxi};
}
