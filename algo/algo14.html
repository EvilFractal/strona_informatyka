vector<int> merger(vector<int> v,int poc,int s,int kon){
    vector<int> pom(kon+1);
    int i=poc,j=s,k=poc;
    while(i<s and j<=kon){
        if(v[i]>v[j]){
            pom[k]=v[j];
            j++;
        }
        else{
            pom[k]=v[i];
            i++;
        }
        k++;
    }
    while(i<s){
        pom[k]=v[i];
        k++;
        i++;
    }
    while(j<=kon){
        pom[k]=v[j];
        k++;
        j++;
    }
    for(int ii=poc;ii<=kon;ii++){
        v[ii]=pom[ii];
    }
    return v;
}
            
vector<int> mergesort(vector<int> v,int poc,int kon){
    if(kon-poc<1) return v;
    if(kon-poc==1){
        if(v[poc]>v[kon]) swap(v[poc],v[kon]);
        return v;
    }
    v=mergesort(v,poc,(poc+kon)/2);
    v=mergesort(v,(poc+kon)/2+1,kon);
    v=merger(v,poc,(poc+kon)/2+1,kon);
    return v;
}
