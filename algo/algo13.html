vector<int> quicksort(vector<int> v,int poc,int kon){
    if(kon-poc<2) return v;
    if(kon-poc==2){
        if(v[poc]>v[poc+1]) swap(v[poc],v[poc+1]);
        return v;
    }
    int piv=v[kon-1],gran=poc;
    for(int i=poc;i<kon;i++){
        if(v[i]<=piv){
            swap(v[i],v[gran]);
            gran++;
        }
    }
    if(gran-poc>1) v=quicksort(v,poc,gran-1);
    if(kon-gran>1) v=quicksort(v,gran,kon);
    return v;
}
