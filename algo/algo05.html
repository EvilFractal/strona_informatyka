string from_dec(int n,int podst){
    string wyn="";
    char z;
    int c;
    while(n){
        c=n%podst;
        if(c>35) z=c+'a'-36;
        else if(c>9) z=c+'A'-10;
        else z=c+'0';
        wyn=z+wyn;
        n/=podst;
    }
    return wyn;
}
            
int to_dec(string s,int podst){
    int wyn=0,curr=1,c;
    for(int i=s.size()-1;i>=0;i--){
        c=(int)s[i];
        if(c>='a') c=c-'a'+36;
        else if(c>='A') c=c-'A'+10;
        else c-='0';
        wyn+=c*curr;
        curr*=podst;
    }
    return wyn;
}
